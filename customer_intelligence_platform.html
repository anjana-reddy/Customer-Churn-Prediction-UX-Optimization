A/B Test Code — 2_analysis.py snippet
# Two-Proportion Z-Test from scipy import stats ctrl=ab[ab.Variant=='Control_A']; var=ab[ab.Variant=='Variant_B'] nA,cA=len(ctrl),ctrl.Converted.sum() nB,cB=len(var), var.Converted.sum() pA,pB=cA/nA, cB/nB lift=(pB-pA)/pA # → +28.3% pp=(cA+cB)/(nA+nB) se=np.sqrt(pp*(1-pp)*(1/nA+1/nB)) z=(pB-pA)/se # → 8.201 pval=1-stats.norm.cdf(z) # → <0.0001 ci_lo=(pB-pA)-1.96*se # → 0.0247 ci_hi=(pB-pA)+1.96*se # → 0.0403 # Decision: REJECT H₀ — Variant B wins print(f"Lift={lift:.1%} Z={z:.3f} p={pval:.4f}")
